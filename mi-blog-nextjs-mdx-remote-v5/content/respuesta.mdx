```csharp
    {

        {
            _connectionString = configuration.GetConnectionString("DefaultConnection")

        public SqlConnection CreateConnection()
        {
            return new SqlConnection(_connectionString);
```

# 5) Repositorio con la conversión del FoxPro a C#

Aquí es donde queda la consulta que en FoxPro se armaba con `TEXT TO cad1`.

## `Repositories/EmpresaRepository.cs`

```csharp
using Microsoft.Data.SqlClient;
using TuProyecto.Data;
using TuProyecto.Dtos;
using TuProyecto.Repositories.Interfaces;

namespace TuProyecto.Repositories
{
    public class EmpresaRepository : IEmpresaRepository
    {
        private readonly DbConnectionFactory _dbConnectionFactory;

        public EmpresaRepository(DbConnectionFactory dbConnectionFactory)
        {
            _dbConnectionFactory = dbConnectionFactory;
        }

        public async Task<EmpresaNitDto?> ObtenerEmpresaPorNitAsync(string nitCed, string tipo)
        {
            const string sql = @"
                SELECT 
                    a.tipodoc,
                    a.regimen,
                    a.nitced,
                    a.dv,
                    a.razon,
                    a.direccion,
                    a.telefono,
                    a.municipio,
                    b.dpto,
                    c.codDpto,
                    b.codmunicipio,
                    c.nombreDpto,
                    b.nombreCiudad,
                    d.num_aprobacion,
                    d.fecha1,
                    d.fecha2,
                    d.numfac1,
                    d.numfac2,
                    d.prefijo
                FROM grl_nita01 a
                LEFT JOIN municipios b ON a.municipio = b.codigo
                LEFT JOIN dpto c ON b.dpto = c.codigo
                LEFT JOIN empresa d ON a.nitced = d.nitced
                WHERE a.nitced = @NitCed
                  AND d.tipo = @Tipo;";

            using var connection = _dbConnectionFactory.CreateConnection();
            using var command = new SqlCommand(sql, connection);

            command.Parameters.AddWithValue("@NitCed", nitCed);
            command.Parameters.AddWithValue("@Tipo", tipo);

            await connection.OpenAsync();

            using var reader = await command.ExecuteReaderAsync();

            if (!await reader.ReadAsync())
                return null;

            return new EmpresaNitDto
            {
                TipoDoc = reader["tipodoc"] == DBNull.Value ? null : reader["tipodoc"].ToString(),
                Regimen = reader["regimen"] == DBNull.Value ? null : reader["regimen"].ToString(),
                NitCed = reader["nitced"] == DBNull.Value ? null : reader["nitced"].ToString(),
                Dv = reader["dv"] == DBNull.Value ? null : reader["dv"].ToString(),
                Razon = reader["razon"] == DBNull.Value ? null : reader["razon"].ToString(),
                Direccion = reader["direccion"] == DBNull.Value ? null : reader["direccion"].ToString(),
                Telefono = reader["telefono"] == DBNull.Value ? null : reader["telefono"].ToString(),
                Municipio = reader["municipio"] == DBNull.Value ? null : reader["municipio"].ToString(),
                Dpto = reader["dpto"] == DBNull.Value ? null : reader["dpto"].ToString(),
                CodDpto = reader["codDpto"] == DBNull.Value ? null : reader["codDpto"].ToString(),
                CodMunicipio = reader["codmunicipio"] == DBNull.Value ? null : reader["codmunicipio"].ToString(),
                NombreDpto = reader["nombreDpto"] == DBNull.Value ? null : reader["nombreDpto"].ToString(),
                NombreCiudad = reader["nombreCiudad"] == DBNull.Value ? null : reader["nombreCiudad"].ToString(),
                NumAprobacion = reader["num_aprobacion"] == DBNull.Value ? null : reader["num_aprobacion"].ToString(),
                Fecha1 = reader["fecha1"] == DBNull.Value ? null : Convert.ToDateTime(reader["fecha1"]),
                Fecha2 = reader["fecha2"] == DBNull.Value ? null : Convert.ToDateTime(reader["fecha2"]),
                NumFac1 = reader["numfac1"] == DBNull.Value ? null : reader["numfac1"].ToString(),
                NumFac2 = reader["numfac2"] == DBNull.Value ? null : reader["numfac2"].ToString(),
                Prefijo = reader["prefijo"] == DBNull.Value ? null : reader["prefijo"].ToString()
            };
        }
    }
}
```

# 7) Servicio

## `Services/EmpresaService.cs`

```csharp
using TuProyecto.Dtos;
using TuProyecto.Repositories.Interfaces;
using TuProyecto.Services.Interfaces;

namespace TuProyecto.Services
{
    public class EmpresaService : IEmpresaService
    {
        private readonly IEmpresaRepository _empresaRepository;

        public EmpresaService(IEmpresaRepository empresaRepository)
        {
            _empresaRepository = empresaRepository;
        }

        public async Task<EmpresaNitDto?> ObtenerEmpresaPorNitAsync(string nitCed, string tipo)
        {
            if (string.IsNullOrWhiteSpace(nitCed))
                throw new ArgumentException("El nitCed es obligatorio.");

            if (string.IsNullOrWhiteSpace(tipo))
                throw new ArgumentException("El tipo es obligatorio.");

            return await _empresaRepository.ObtenerEmpresaPorNitAsync(nitCed, tipo);
        }
    }
}
```

# 8) Controller

## `Controllers/EmpresaController.cs`

```csharp
using Microsoft.AspNetCore.Mvc;
using TuProyecto.Services.Interfaces;

namespace TuProyecto.Controllers
{
    [ApiController]
    [Route("api/[controller]")]
    public class EmpresaController : ControllerBase
    {
        private readonly IEmpresaService _empresaService;

        public EmpresaController(IEmpresaService empresaService)
        {
            _empresaService = empresaService;
        }

        [HttpGet("{nitCed}")]
        public async Task<IActionResult> ObtenerPorNit(string nitCed, [FromQuery] string tipo = "1")
        {
            var empresa = await _empresaService.ObtenerEmpresaPorNitAsync(nitCed, tipo);

            if (empresa == null)
                return NotFound(new
                {
                    mensaje = "No se encontró información para el NIT consultado."
                });

            return Ok(empresa);
        }
    }
}
```

```csharp
using TuProyecto.Data;
using TuProyecto.Repositories;
using TuProyecto.Repositories.Interfaces;
using TuProyecto.Services;
using TuProyecto.Services.Interfaces;

var builder = WebApplication.CreateBuilder(args);

builder.Services.AddSwaggerGen();

{
    app.UseSwagger();
    app.UseSwaggerUI();
}

app.UseHttpsRedirection();
app.UseAuthorization();
app.MapControllers();

app.Run();
```

# 13) Mejora importante: evitar `AddWithValue`

Aunque el código anterior funciona, en proyectos reales yo prefiero dejar los tipos definidos para evitar problemas de conversión y rendimiento.

## Versión mejorada del agregado de parámetros

Dentro del repositorio cambia esto:

```csharp
command.Parameters.AddWithValue("@NitCed", nitCed);
command.Parameters.AddWithValue("@Tipo", tipo);
```

por esto:

```csharp
command.Parameters.Add("@NitCed", System.Data.SqlDbType.VarChar, 20).Value = nitCed;
command.Parameters.Add("@Tipo", System.Data.SqlDbType.VarChar, 5).Value = tipo;
```

# 14) Mi recomendación para dejarlo más reutilizable

En vez de dejar el `tipo = "1"` fijo dentro del SQL, déjalo como parámetro como ya hicimos, porque así el mismo método te sirve para otros casos.

Y si quieres llevarlo a un estilo más limpio todavía, puedes hacer que el controller reciba ambos parámetros:

```csharp
[HttpGet]
public async Task<IActionResult> Obtener([FromQuery] string nitCed, [FromQuery] string tipo)
{
    var empresa = await _empresaService.ObtenerEmpresaPorNitAsync(nitCed, tipo);

    if (empresa == null)
        return NotFound();

    return Ok(empresa);
}
```

Llamándolo así:

```http
GET /api/empresa?nitCed=890116102&tipo=1
```
